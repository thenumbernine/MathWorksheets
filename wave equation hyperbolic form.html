$g^{00} \Phi_{,00} + 2 g^{0i} \Phi_{,0i} + g^{ij} \Phi_{,ij} - g^{\mu\nu} {\bar{\Gamma}^0}_{\mu\nu} \Phi_{,0} - g^{\mu\nu} {\bar{\Gamma}^i}_{\mu\nu} \Phi_{,i} = 0$

$-\alpha^{-2} \Phi_{,00} + 2 \alpha^{-2} \beta^{i} \Phi_{,0i} + (\gamma^{ij} - \alpha^{-2} \beta^i \beta^j) \Phi_{,ij} - \bar{\Gamma}^0 \Phi_{,0} - \bar{\Gamma}^i \Phi_{,i} = 0$
$\Phi_{,00} - 2 \beta^{i} \Phi_{,0i} + (\beta^i \beta^j - \alpha^2 \gamma^{ij}) \Phi_{,ij} + \alpha^2 \bar{\Gamma}^0 \Phi_{,0} + \alpha^2 \bar{\Gamma}^i \Phi_{,i} = 0$

Let $\Pi = n^\mu \Phi_{,\mu}$
$\Pi = n^0 \Phi_{,0} + n^i \Phi_{,i}$
$\Pi = \frac{1}{\alpha} (\Phi_{,0} - \beta^i \Phi_{,i})$

So $\Phi_{,0} - \beta^j \Phi_{,j} = \alpha \Pi$
Let $\frac{d}{dx^0} = \partial_0 - \mathcal{L}_\vec\beta$
So $\frac{d}{dx^0} \Phi = \alpha \Pi$

Spatial derivative of $\Phi_{,0}$:
$\Phi_{,0i} = {\beta^j}_{,i} \Phi_{,j} + \beta^j \Phi_{,ij} + \alpha_{,i} \Pi + \alpha \Pi_{,i}$

Derivative of $\Pi$:
$\Pi_{,\mu} = -\frac{1}{\alpha^2} \alpha_{,\mu} (\Phi_{,0} - \beta^j \Phi_{,j}) + \frac{1}{\alpha} ( \Phi_{,0\mu} - {\beta^j}_{,\mu} \Phi_{,j} - \beta^j \Phi_{,j \mu} )$
$\Pi_{,\mu} = \frac{1}{\alpha} ( \Phi_{,0\mu} - \alpha_{,\mu} \Pi - {\beta^j}_{,\mu} \Phi_{,j} - \beta^j \Phi_{,j \mu} )$

spatial deriative of $\Pi$:
$\Pi_{,i} = \frac{1}{\alpha} ( \Phi_{,0i} - \alpha_{,i} \Pi - {\beta^j}_{,i} \Phi_{,j} - \beta^j \Phi_{,j i} )$

$\Phi_{,0i} = \alpha \Pi_{,i} + \alpha_{,i} \Pi + {\beta^j}_{,i} \Phi_{,j} + \beta^j \Phi_{,j i} $

timelike derivative of $\Pi$:
$\Pi_{,0} = \frac{1}{\alpha} ( \Phi_{,00} - \alpha_{,0} \Pi - {\beta^j}_{,0} \Phi_{,j} - \beta^j \Phi_{,j 0} )$

$\Psi^i = \gamma^{ij} \Phi_{,j}$

${\Psi^i}_{,\alpha} = (\gamma^{ij} \Phi_{,j})_{,\alpha}$

${\Psi^i}_{,0} = -( \gamma_{mn,k} \beta^k + \gamma_{mk} {\beta^k}_{,n} + \gamma_{kn} {\beta^k}_{,m} - 2 \alpha K_{mn} )\gamma^{im} \gamma^{nj} \Phi_{,j} + \gamma^{ij} \Pi_{,j}$
